\begin{usecase}

\addtitle{Használati eset:}{Input elem hozzáadása} 
\addfield{Szint:}{Input}
\addfield{Elsődleges aktor:}{Felhasználó}
\additemizedfield{Szereplők és érdekeik:}{
	\item Felhasználó: Egy input elemet hozzáadjon a felhasználó a feldolgozási sorhoz.
}

\additemizedfield{Elő feltételek:}{
	\item A rendszer álljon készen az input beolvasására: létezzen input lista, és ne legyen folyamatban feldolgozás.
} 


\additemizedfield{Sikeres végrehajtás (utófeltételek):}{
	\item Az input listához az új elem hozzáadásra került.
	\item A rendszer kezdeményezi a kibővített inputlista megjelenítését
}

\addscenario{Fő (sikeres) szcenárió:}{
	\item A felhasználó jelzi szándékát, hogy bővíteni kívánja az input listát egy új elemmel.
	\item A felhasználó kiválasztja az új input elemnek szánt adatforrást.
	\item A rendszer azonosítja az adatforrás formátumát.
	\item A rendszer beolvassa az adatsort az adatforrásból.
	\item A rendszer létrehozza az új input elemet és hozzáfűzi az inputlistához.
}

\addscenario{Alternatív szcenáriók:}{
	\item[3.a] Nem sikerült hozzáférni az adatforráshoz: nincs jogosultság
		\begin{enumerate}
		\item[1.] A rendszer hibaüzenetet küld a felhasználónak, a sikertelen műveletről.
		\end{enumerate}
		
	\item[3.b] Nem sikerült azonosítani az adatforrás formátumát
		\begin{enumerate}
		\item[1.] A rendszer hibaüzenetet küld a felhasználónak, a sikertelen műveletről.
		\end{enumerate}
		
		
	\item[3.c] Nem támogatott adatforrás formátum
		\begin{enumerate}
		\item[1.] A rendszer hibaüzenetet küld a felhasználónak, a sikertelen műveletről.
		\end{enumerate}
				
}

\end{usecase}
